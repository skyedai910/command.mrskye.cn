解密 session

# 解密json键值对没有base64解码，还是使用P神的解密脚本
# 自行选择python2/3
python(2/3) flask_session_cookie_manager(2/3).py decode -c <session>

加密 session

# 获取SECRET_KEY前提下，用于伪造session
# session内json键值对需要是明文无编码
# 自行选择python2/3
python(2/3) flask_session_cookie_manager(2/3).py encode -s "SECRET_KEY" -t "会话信息"
